Current flow (blocks 100-300ms):

3. Sharp optimization (BLOCKING - lines 53-70)
4. Upload to MinIO

New flow (instant response):
1. Validate file

3. Upload ORIGINAL to MinIO immediately
4. Return response with URL
5. Schedule optimization in after() callback

Import after() from next/server:
```typescript
import { NextRequest, NextResponse, after } from "next/server";
```

Replace lines 48-70 with:
```typescript
// Upload original immediately (no processing)
const buffer = Buffer.from(await file.arrayBuffer());
const fileExtension = file.type.split("/")[1];
const uniqueFilename = `${randomUUID()}.${fileExtension}`;
const filepath = `${folder}/${uniqueFilename}`;

await minioClient.putObject(
  BUCKETS.NEWS_IMAGES,
  filepath,
  buffer,
  buffer.length,
  {
    "Content-Type": file.type,
    "Original-Filename": file.name,
  }
);

const url = await minioClient.presignedGetObject(
  BUCKETS.NEWS_IMAGES,
  filepath,
  7 * 24 * 60 * 60
);

// Schedule optimization in background (non-blocking)
after(async () => {
  try {
    // Only optimize large images (>1MB)
    if (file.size > 1024 * 1024) {
      const image = sharp(buffer);
      const metadata = await image.metadata();

      let optimizedBuffer: Buffer;
      if (metadata.width && metadata.width > 1920) {
        optimizedBuffer = await image
          .resize(1920, null, { withoutEnlargement: true })
          .jpeg({ quality: 85 })
          .toBuffer();
      } else {
        optimizedBuffer = await image
          .jpeg({ quality: 85 })
          .toBuffer();
      }

      // Replace original with optimized version
      await minioClient.putObject(
        BUCKETS.NEWS_IMAGES,
        filepath,
        optimizedBuffer,
        optimizedBuffer.length,
        { "Content-Type": "image/jpeg" }
      );

      console.log(`Optimized ${filepath}: ${buffer.length} -> ${optimizedBuffer.length} bytes`);
    }
  } catch (error) {
    console.error("Background optimization failed:", error);
    // Don't throw - response already sent
  }
});

return NextResponse.json({
  url,
  filepath,
  size: buffer.length,
  mimeType: file.type,
});
```

Important:
- Import `after` from 'next/server'
- Remove the synchronous sharp processing block
- The response returns the original file URL immediately
- Background optimization replaces the file in-place (same path)
- Always use async function inside after() and await all promises
- Add try-catch inside after() callback - errors cannot be sent to client

Modify executeBatchDocuments function to return file paths:
```typescript
async function executeBatchDocuments(tx: any, action: string, id: string): Promise<string | null> {
  switch (action) {
    case "delete":
      // Fetch document to get file path BEFORE deletion
      const document = await tx.document.findUnique({
        where: { id },
        select: { id: true, filePath: true },
      });

      if (!document) {
        throw new Error("Documento non trovato");
      }

      // Delete from database
      await tx.document.delete({ where: { id } });

      // Return file path for cleanup outside transaction
      return document.filePath || null;

    default:
      throw new Error("Azione non supportata per documenti");
  }
}
```

Update the POST handler to collect file paths and cleanup:
```typescript
// Inside POST handler, before the transaction
const filesToDelete: string[] = [];

// Inside the transaction loop for documents case:
case "documents":
  const filePath = await executeBatchDocuments(tx, action, id);
  if (filePath) {
    filesToDelete.push(filePath);
  }
  break;

// After the transaction closes, before return:
if (filesToDelete.length > 0) {
  after(async () => {
    try {
      await minioClient.removeObjects(BUCKETS.DOCUMENTS, filesToDelete);
      console.log(`MinIO cleanup: deleted ${filesToDelete.length} files`);
    } catch (error) {
      console.error("MinIO cleanup failed:", error);
      // Log for manual cleanup but don't fail request
    }
  });
}
```

- Collect file paths during transaction but cleanup AFTER transaction commits
- Use after() for non-blocking cleanup
- Check filesToDelete.length > 0 before calling removeObjects (empty array causes error)
