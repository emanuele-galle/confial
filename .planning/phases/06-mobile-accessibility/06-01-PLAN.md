**Framer Motion implementation:**
- Use `motion.div` with `drag="x"` for horizontal swipe
- Drag constraints: `dragConstraints={{ left: -280, right: 0 }}`
- Sidebar width: 280px (w-70)
- `onDragEnd` handler: if dragged more than 100px left, close sidebar
- Animate open/close with `animate={{ x: isOpen ? 0 : -280 }}`
- Backdrop overlay with fade animation

**Gesture detection:**
```typescript
const handleDragEnd = (event: any, info: PanInfo) => {
  if (info.offset.x < -100) {
    onClose();
  }
};
```

**Visual effects:**
- Sidebar slides in from left
- Semi-transparent backdrop (bg-black/50) when open
- Tap backdrop to close
- Spring animation for natural feel: `transition={{ type: "spring", damping: 25 }}`

```tsx
export default function DashboardLayout({ children }) {
  const [sidebarOpen, setSidebarOpen] = useState(false);

  return (
    <div className="min-h-screen bg-gray-50">
      {/* Desktop sidebar - hidden on mobile */}
      <aside className="hidden lg:fixed lg:inset-y-0 lg:flex lg:w-64 lg:flex-col">
        <DashboardSidebar />
      </aside>

      {/* Mobile swipeable sidebar */}
      <div className="lg:hidden">
        <SwipeableSidebar isOpen={sidebarOpen} onClose={() => setSidebarOpen(false)}>
          <DashboardSidebar onNavClick={() => setSidebarOpen(false)} />
        </SwipeableSidebar>
      </div>

      {/* Main content with responsive padding */}
      <div className="lg:pl-64">
        <DashboardHeader onMenuClick={() => setSidebarOpen(true)} />
        <main className="p-4 sm:p-6 lg:p-8 pb-20 lg:pb-8">
          {children}
        </main>
      </div>

      {/* Mobile bottom nav */}
      <MobileBottomNav />
    </div>
  );
}
```
