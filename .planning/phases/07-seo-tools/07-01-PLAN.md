```typescript
export function calculateFleschIT(text: string): {
  score: number;
  level: "molto_facile" | "facile" | "normale" | "difficile" | "molto_difficile";
  description: string;
} {
  // Italian Flesch formula: 206 - (0.65 * avgSyllables) - (avgWordsPerSentence)
  // Simplified for Italian: 206.835 - (1.015 * avgWordsPerSentence) - (84.6 * avgSyllablesPerWord)

  const sentences = text.split(/[.!?]+/).filter(s => s.trim());
  const words = text.split(/\s+/).filter(w => w.match(/\w/));
  const syllables = words.reduce((sum, word) => sum + countSyllablesIT(word), 0);

  const avgWordsPerSentence = words.length / Math.max(sentences.length, 1);
  const avgSyllablesPerWord = syllables / Math.max(words.length, 1);

  const score = 206.835 - (1.015 * avgWordsPerSentence) - (84.6 * avgSyllablesPerWord);

  // Clamp to 0-100 range
  const clampedScore = Math.max(0, Math.min(100, score));

  return {
    score: Math.round(clampedScore),
    level: getFleschLevel(clampedScore),
    description: getFleschDescription(clampedScore),
  };
}

function countSyllablesIT(word: string): number {
  // Italian syllable counting: count vowel groups
  const vowels = word.toLowerCase().match(/[aeiouàèéìòù]+/g);
  return vowels ? vowels.length : 1;
}
```

**Keyword density visualization (SEO-03):**
```typescript
export function getKeywordDensity(text: string, minLength = 3): {
  keywords: { word: string; count: number; density: number }[];
  totalWords: number;
} {
  // Remove HTML tags and normalize
  const cleanText = text.replace(/<[^>]+>/g, ' ').toLowerCase();
  const words = cleanText.split(/\s+/).filter(w => w.length >= minLength && !isStopWord(w));

  const wordCount: Record<string, number> = {};
  words.forEach(w => { wordCount[w] = (wordCount[w] || 0) + 1; });

  const keywords = Object.entries(wordCount)
    .map(([word, count]) => ({
      word,
      count,
      density: (count / words.length) * 100,
    }))
    .sort((a, b) => b.count - a.count)
    .slice(0, 20); // Top 20 keywords

  return { keywords, totalWords: words.length };
}

const ITALIAN_STOP_WORDS = new Set([
  'il', 'lo', 'la', 'i', 'gli', 'le', 'un', 'uno', 'una',
  'di', 'a', 'da', 'in', 'con', 'su', 'per', 'tra', 'fra',
  'e', 'o', 'ma', 'non', 'che', 'se', 'come', 'quando',
  'questo', 'quello', 'essere', 'avere', 'fare', 'dire',
  'anche', 'più', 'solo', 'così', 'già', 'ancora', 'molto',
]);

function isStopWord(word: string): boolean {
  return ITALIAN_STOP_WORDS.has(word);
}
```

Note: Use simple n-gram extraction instead of compromise.js to avoid large dependency
```typescript
export function extractPhrases(text: string, n = 2): {
  phrases: { phrase: string; count: number }[];
} {
  const cleanText = text.replace(/<[^>]+>/g, ' ').toLowerCase();
  const words = cleanText.split(/\s+/).filter(w => w.match(/\w{3,}/));

  // Extract n-grams (bigrams and trigrams)
  const phrases: Record<string, number> = {};

  for (let i = 0; i < words.length - n + 1; i++) {
    const phrase = words.slice(i, i + n).join(' ');
    // Skip if any word is a stop word
    if (words.slice(i, i + n).some(w => isStopWord(w))) continue;
    phrases[phrase] = (phrases[phrase] || 0) + 1;
  }

  return {
    phrases: Object.entries(phrases)
      .filter(([_, count]) => count >= 2) // At least 2 occurrences
      .map(([phrase, count]) => ({ phrase, count }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 10),
  };
}
```
